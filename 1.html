import turtle

name = turtle.Turtle() # Creating a turtle object named 'name'
name.fd(100) # Moves the turtle by 100px in forward direction
name.bk(200) # Moves the turtle by 200px in backward direction
name.lt(90) # Turns the turtle by 90 degrees towards its left
name.forward(100) # Moves the turtle by 100px in forward direction
name.rt(45) # Turns the turtle by 90 degrees towards its right
name.forward(100) # Moves the turtle by 100px in forward direction

name.goto(100,100) # Moves the turtle to the co-ordinate (100,100)
name.setpos(100,200) # Moves the turtle to the co-ordinate (100,200)
name.setposition(200,200) # Moves the turtle to the co-ordinate (200,200)

name.setx(500) # Changes the x-coordinate of the turtle to 500 keeping y-coordinate constant
name.sety(500) # Changes the y-coordinate of the turtle to 500 keeping x-coordinate constant
name.seth(120) # Rotates the turtle through 120 degrees
name.forward(100) # Moves the turtle by 100px in forward direction

name.circle(100) # Draws a circle of radius 100px
name.goto(200,200) # Moves the turtle to the co-ordinate (200,200)
name.dot(200) # Draws a dot of diameter 200px

turtle.done()
